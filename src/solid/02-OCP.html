class Output {

    public function __construct(array $data) {
        $this->data = $data
    }

    public function render($format = 'html') {
        $out = '';
        switch ($format) {
            case 'html':
                $out = \View::make('article', $this->getData());
                break;
            case 'json':
                $out = json_encode($this->getData());
                break;
            default:
                throw new Exception("Unexpected format [$format]");
        }
        return $out;
    }
}
